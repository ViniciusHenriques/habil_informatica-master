 function IssRetido: boolean;

    Result := AnsiIndexText(CampoTecno('IssRetido'), ['1', 'sim', 'true', 'Sim']) <> -1;
  end;

begin
  CODIGOS_CIDADES[0] := '4318705';
  CODIGOS_CIDADES[1] := '4312401';

  CODIGOS_CIDADES[4] := '4307708';

  CODIGOS_CIDADES[6] := '4314100';
  CODIGOS_CIDADES[7] := '4303103';
  CODIGOS_CIDADES[8] := '4321204';
  CODIGOS_CIDADES[9] := '4317608';
  CODIGOS_CIDADES[10] := '3303906';
  CODIGOS_CIDADES[11] := '4312658';
  NOMES_CIDADES[0] := 'SAOLEOPOLDO';
  NOMES_CIDADES[1] := 'MONTENEGRO';
  NOMES_CIDADES[2] := 'LAJEADORS';
  NOMES_CIDADES[3] := 'VENANCIOAIRESRS';
  NOMES_CIDADES[4] := 'ESTEIORS';

  NOMES_CIDADES[8] := 'TAQUARARS';

  NOMES_CIDADES[10] := 'GASPARSC';
  NOMES_CIDADES[11] := 'NAOMETOQUERS';

  Progress('Criando XSDDataset');
  CriarDatasets;

  _Tx2 := TStringList.Create;
  _CamposMunicipio := TStringList.Create;
  
  try
    Progress('Carregando conteúdo/arquivo TX2');
    CarregarTX2;

    _DsTecnoAtual := nil;

    if _PadraoTecnoNFSe then
    begin
      Progress('Usando Padrão TecnoNFSe: preparando datasets');
      PrepararTecnoNFSeDatasets;

      // Passar dados para o Datasets, convertendo do padrão TecnoNFSe para o padrão da Cidade
      Datasets.GetDataSetByName(DSCIDADE_LOTE).Insert;

      // Começar com as notas
      Progress('Usando Padrão TecnoNFSe: carregando dados das notas');
      _DsTecnoAtual := TecnoNFSeDS.GetDataSetByName(DSTECNO_NOTA);
      _DsCidadeAtual := Datasets.GetDataSetByName(DSCIDADE_NOTA);

      _QtdeRps := _DsTecnoAtual.RecordCount;

      _DsTecnoAtual.First;

      while not _DsTecnoAtual.Eof do
      begin
        _DsCidadeAtual.Append;

        if CampoTecno('IdRps') <> '' then
          SetarCampo('IdRps')
        else
        begin
          _ValorCampo := 'RPS_' + CampoTecno('NumeroRps') + '_' + CampoTecno('SerieRps') + '_' + CampoTecno('TipoRps');
          SetarCampoValor('IdRps', _ValorCampo);
        end;

        _ValorCampo := CampoTecno('NumeroRps');
        VerificaSeEhNumero('NumeroRps', _ValorCampo);
        SetarCampoObrigatorioValorTamanho('NumeroRps','NumeroRps', _ValorCampo, 15);

        SetarCampoObrigatorioTamanho('SerieRps','SerieRps', 5);

        _ValorCampo := CampoTecno('TipoRps');
        if AnsiMatchText(_ValorCampo, ['1', '2', '3']) then
          SetarCampoObrigatorioValorTamanho('TipoRps', 'TipoRps', _ValorCampo, 1)
        else
          RaiseException('Para essa cidade o campo TipoRps deve ser informado da seguinte forma: 1 - RPS; 2 - Nota Fiscal Conjugada (Mista); 3 – Cupom.');

        _ValorCampo := FormatAnyDateTime(CampoTecno('DataEmissao'), XSDDATETIMEFORMAT);
        SetarCampoValorTamanhoObrigatorio('DataEmissao', _ValorCampo, 23);

        SetarCampoValorTamanhoObrigatorio('OptanteSimplesNacional', CampoTecno('OptanteSimplesNacional'), 1);
        SetarCampoValorTamanho('IncentivadorCultural', CampoTecno('IncentivadorCultural'), 1);

        SetarCampoValorTamanhoObrigatorio('Status', CampoTecno('SituacaoNota'), 1);

        _CidadeConfigurada := Params.Values['Cidade'];

        for i := 0 to 11 do
        begin
          if SameText(_CidadeConfigurada, NOMES_CIDADES[i]) then
            _IndexCidadeConfig := i;
        end;

        if _IndexCidadeConfig >= 0 then
          _CodigoCidadeConfig := CODIGOS_CIDADES[_IndexCidadeConfig];

        case _IndexCidadeConfig of
          0:
          begin

            // [5.0] Prestação dentro ou fora do municipio para São Leopoldo
            case StrToIntDef(CampoTecno('TipoTributacao'), -1) of
            6:
            begin
              if CampoTecno('ISSRetido') = '2' then
                SetarCampoValor('NaturezaOperacao', '52') // 5.2 - Imposto devido em São Leopoldo, sem obrigação de retenção na fonte. - Pessoa Física
              else
                SetarCampoValor('NaturezaOperacao', '51'); // 5.1 - Imposto devido em São Leopoldo, com obrigação de retenção na fonte. - Pessoa Jurídica
            end;
            7:
              if (_CodigoCidadeConfig <> CampoTecno('CodigoCidadePrestacao')) then
                SetarCampoValor('NaturezaOperacao', '63'); // 6.3 - Imposto devido fora de São Leopoldo
            end;


            if (CampoTecno('OptanteSimplesNacional') = '1') or (CampoTecno('RegimeEspecialTributacao') = '5') then
              SetarCampoValor('NaturezaOperacao', '59'); // 5.9 - Imposto recolhido pelo regime único de arrecadação - Simples Nacional e MEI
            if (CampoTecno('NaturezaTributacao') = '2') or (AnsiMatchText(CampoTecno('TipoTributacao'), ['1', '2', '4'])) then
              SetarCampoValor('NaturezaOperacao', '58'); // 5.8 - Não tributável, prestador com ISS Fixo, Imunidade, Isenção ou Não Incidência.

            // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') and (not AnsiMatchText(CampoCidade('NaturezaOperacao'), ['52', '58'])) then
                SetarCampoValor('NaturezaOperacao', '78'); // 7.8 - Não tributável, prestação no exterior.
            end;
          end;

          1:
          begin
            if CampoTecno('CodigoCidadeTomador') = _CodigoCidadeConfig then
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '51') // 5.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '52'); // 5.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end

              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '59') // 5.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional.
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '58'); // 5.8 - Não tributável.

            end
            else
            // [6.0] Prestação de serviços para outros municípos da federação.
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '61') // 6.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '68') // 6.8 - Não tributável.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '69') // 6.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional.
              else if IssRetido then
                SetarCampoValor('NaturezaOperacao', '63') // 6.3 - Imposto devido fora de <cidade>, com obrigação de retenção na fonte.
              else
                SetarCampoValor('NaturezaOperacao', '64'); // 6.4 - Imposto devido fora de <cidade>, sem obrigação de retenção na fonte.
            end;
            // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
              begin
                if CampoTecno('TipoTributacao') = '4' then
                  SetarCampoValor('NaturezaOperacao', '78'); // 7.8 - Não tributável.
                if (CampoTecno('OptanteSimplesNacional') = '1') and (SameText(_CidadeConfigurada, 'SaoLeopoldo')) then
                  SetarCampoValor('NaturezaOperacao', '79'); // 7.9 - Imposto recolhido pelo regime único de arrecadação.
              end;
            end;
          end;

          2:
          begin
            if CampoTecno('CodigoCidadePrestacao') = _CodigoCidadeConfig then
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '51') // 5.1 - Imposto Devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '52'); // 5.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end;
              
              
              
              if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '58') // 5.8 - Não Tributável.
              else if (CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig) then
                SetarCampoValor('NaturezaOperacao', '59') // 5.9 - Imposto recolhido pelo regime único de arrecadação.
              else if (CampoTecno('RegimeEspecialTributacao') = '5') and (Not IssRetido) then
                SetarCampoValor('NaturezaOperacao', '56') // 5.6 - Microempreendedor individual-ISSQN recolhido através de cota fixa mensal, não sujeito a retenção.
              else if CampoTecno('NaturezaTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '57'); // 5.7 - ISSQN Fixo.
              
            end
            else
            begin
              // [6.0] Prestação de serviços para outros municípos da federação.
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '61') // 6.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if IssRetido then
                SetarCampoValor('NaturezaOperacao', '63') // 6.3 - Imposto devido fora de <cidade> com obrigação de retenção na fonte.
              else
                SetarCampoValor('NaturezaOperacao', '64'); // 6.4 - Imposto devido fora de <cidade>, sem obrigação de retenção na fonte.
              
              
              
              if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '68') // 6.8 - Não tributável.
              else if (CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig) then
                SetarCampoValor('NaturezaOperacao', '69') // 6.9 - Imposto recolhido pelo regime único de arrecadação.
              else if (CampoTecno('RegimeEspecialTributacao') = '5') and (Not IssRetido) then
                SetarCampoValor('NaturezaOperacao', '66') // 6.6 - Microempreendedor individual-ISSQN recolhido através de cota fixa mensal, não sujeito a retenção.
              else if CampoTecno('NaturezaTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '67'); // 6.7 - ISSQN Fixo.
              
            end;
            // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
              begin
                if CampoTecno('TipoTributacao') = '4' then
                begin
                  if CampoTecno('ExigibilidadeISS') = '4' then
                    SetarCampoValor('NaturezaOperacao', '72') // 7.2 -  Exportação de serviço - Não tributável.
                  else
                    SetarCampoValor('NaturezaOperacao', '78'); // 7.8 - Não tributável.
                end
                else if (CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig) then
                  SetarCampoValor('NaturezaOperacao', '79'); // 7.9 - Imposto recolhido pelo regime único de arrecadação.
              end;
            end;
          end;

          3, 4, 5:
          begin
            if CampoTecno('CodigoCidadeTomador') = _CodigoCidadeConfig then
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '51') // 5.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '52'); // 5.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if ((CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('TipoTributacao') <> '4')) then
                SetarCampoValor('NaturezaOperacao', '59') // 5.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional.
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '58'); // 5.8 - Não tributável.

            end
            else
            // [6.0] Prestação de serviços para outros municípos da federação.
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '61') // 6.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '68') // 6.8 - Não tributável.
              else if IssRetido then
                SetarCampoValor('NaturezaOperacao', '63') // 6.3 - Imposto devido fora de <cidade>, com obrigação de retenção na fonte.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '69') // 6.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional.
              else
                SetarCampoValor('NaturezaOperacao', '64'); // 6.4 - Imposto devido fora de <cidade>, sem obrigação de retenção na fonte.
            end;
            // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
              begin
                if CampoTecno('TipoTributacao') = '4' then
                  SetarCampoValor('NaturezaOperacao', '78') // 7.8 - Não tributável.
                else if CampoTecno('OptanteSimplesNacional') = '1' then
                  SetarCampoValor('NaturezaOperacao', '79'); // 7.9 - Imposto recolhido pelo regime único de arrecadação.
              end;
            end;
          end;

          6:
          begin
            if CampoTecno('CodigoCidadeTomador') = _CodigoCidadeConfig then
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                begin
                  if CampoTecno('OptanteSimplesNacional') = '1' then
                    SetarCampoValor('NaturezaOperacao', '54') //5.4 Imposto devido a <cidade>, com obrigação de retenção na fonte, tributado de acordo com o regime único de arrecadação.
                  else
                    SetarCampoValor('NaturezaOperacao', '51'); // 5.1 Imposto Devido em <cidade>, com obrigação de retenção na fonte.
                end
                else
                  SetarCampoValor('NaturezaOperacao', '52'); //5.2 Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if CampoTecno('TipoTributacao') = '1' then
                SetarCampoValor('NaturezaOperacao', '55') // 5.5 Isento.
              else if CampoTecno('TipoTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '56') // 5.6 Imune.
              else if CampoTecno('NaturezaTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '57') // 5.7 Exigibilidade suspensa por decisão judicial.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '53') // 5.3 Imposto recolhido pelo Regime único de Arrecadação (Simples Nacional).
              else if CampoTecno('NaturezaTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '58'); // 5.8 Imposto recolhido com valor fixo.
            end
            else
            // [6.0] Prestação de serviços para outros municípos da federação.
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '61') // 6.1 Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if CampoTecno('TipoTributacao') = '1' then
                SetarCampoValor('NaturezaOperacao', '65') // 6.5 Isento.
              else if CampoTecno('TipoTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '66') // 6.6 Imune.
              else if CampoTecno('NaturezaTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '67') // 6.7 Exigibilidade suspensa por decisão judicial.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '63') // 6.3 Imposto recolhido pelo Regime único de Arrecadação (Simples Nacional).
              else if CampoTecno('NaturezaTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '68') // 6.8 Imposto recolhido com valor fixo.
              else
                SetarCampoValor('NaturezaOperacao', '64'); // 6.4 Imposto devido fora de <cidade>.
            end;
            // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
              begin
                if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
                  SetarCampoValor('NaturezaOperacao', '71'); // 7.1 Imposto devido a <cidade>.
                if CampoTecno('TipoTributacao') = '4' then
                  SetarCampoValor('NaturezaOperacao', '72'); // 7.2 Não tributável.
              end;
            end;
          end;

          7:
          begin
          //Manual Chachoeirinha 07/04/2019 http://www.cachoeirinha.rs.gov.br/portal/attachments/article/1623/Tabela%20de%20C%C3%B3digo%20de%20Opera%C3%A7%C3%B5es%20para%20Prestadores%20de%20Servi%C3%A7os%20Cachoeirinha.pdf
            if CampoTecno('CodigoCidadePrestacao') = _CodigoCidadeConfig then
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '53') // 5.3 - Imposto devido em Cachoeirinha, com obrigação de retenção na fonte
                                                            //  Classificam-se neste código as prestações de serviços que o
                                                            //  estabelecimento tomador esteja localizado na da cidade de Cachoeirinha, cujo imposto
                                                            //  seja devido no município de Cachoeirinha e a obrigação de recolhimento do ISS seja do
                                                            //  contratante do serviço.
                else
                  SetarCampoValor('NaturezaOperacao', '52'); // 5.2 - Imposto devido em Cachoeirinha, sem obrigação de retenção na fonte Classificam-se
                                                             // neste código as prestações de serviços realizados na cidade de Cachoeirinha e
                                                             // cuja obrigação de recolhimento do ISS seja do prestador do serviço.
              end;
              if AnsiMatchText(CampoTecno('TipoTributacao'),['1','2','4']) then
                SetarCampoValor('NaturezaOperacao', '58') // 5.8 - Não Tributável.
              else if (CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('NaturezaTributacao') = '1') then
                SetarCampoValor('NaturezaOperacao', '59'); // 5.9 - Imposto recolhido pelo regime único de arrecadação.
              
            end
            else
            // [6.0] Prestação de serviços para outros municípos da federação.
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if not IssRetido then
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '63') // 6.3 - Imposto devido fora de <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '64'); // 6.4 - Imposto devido fora de <cidade>, sem obrigação de retenção na fonte.
                if AnsiMatchText(CampoTecno('TipoTributacao'),['1','2','4']) then
                  SetarCampoValor('NaturezaOperacao', '68'); // 6.8 - Não tributável.
                if (CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('NaturezaTributacao') = '1') then
                  SetarCampoValor('NaturezaOperacao', '69'); // 6.9 - Imposto recolhido pelo regime único de arrecadação.
              end;
            end;
            // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
              begin
                if AnsiMatchText(CampoTecno('TipoTributacao'),['1','2','4']) then
                  SetarCampoValor('NaturezaOperacao', '78') // 7.8 - Não tributável.
                else if (CampoTecno('OptanteSimplesNacional') = '1') and (CampoTecno('NaturezaTributacao') = '1') then
                  SetarCampoValor('NaturezaOperacao', '79'); // 7.9 - Imposto recolhido pelo regime único de arrecadação.
              end;
            end;
          end;

          8:
          begin
            if CampoTecno('CodigoCidadeTomador') = _CodigoCidadeConfig then
            begin
              if (CampoTecno('OptanteSimplesNacional') = '1') and (IssRetido) then
                SetarCampoValor('NaturezaOperacao', '57') // 5.7 - Imposto recolhido pelo regime único de arrecadação Simples Nacional com obrigação de retenção.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '59') // 5.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional.
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '58') // 5.8 - Não tributável.
              else if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '51') // 5.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '52'); // 5.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end;

            end
            else
            // [6.0] Prestação de serviços para outros municípos da federação.
            begin
              if (CampoTecno('OptanteSimplesNacional') = '1') and (IssRetido) then
                SetarCampoValor('NaturezaOperacao', '67') // 6.7 - Imposto recolhido pelo regime único de arrecadação Simples Nacional com obrigação de retenção.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
                SetarCampoValor('NaturezaOperacao', '69') // 6.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional.
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '68') // 6.8 - Não tributável.
              else if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if IssRetido then
                  SetarCampoValor('NaturezaOperacao', '61') // 6.1 - Imposto devido em <cidade>, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 - Imposto devido em <cidade>, sem obrigação de retenção na fonte.
              end
              else if IssRetido then
                SetarCampoValor('NaturezaOperacao', '63') // 6.3 - Imposto devido fora de <cidade>, com obrigação de retenção na fonte.
              else
                SetarCampoValor('NaturezaOperacao', '64'); // 6.4 - Imposto devido fora de <cidade>, sem obrigação de retenção na fonte.
            end;
              // [7.0] Prestação de serviços para o exterior.
            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
              begin
                if CampoTecno('TipoTributacao') = '4' then
                  SetarCampoValor('NaturezaOperacao', '78'); // 7.8 - Não tributável.
              end;
            end;
          end;

          9, 10, 11:
          begin
            if CampoTecno('CodigoCidadePrestacao') = _CodigoCidadeConfig then
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if CampoTecno('ISSRetido') = '1' then
                  SetarCampoValor('NaturezaOperacao', '51') // 5.1 - Imposto Devido em Santo Antônio da Patrulha, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '52'); // 5.2 - Imposto devido em Santo Antônio da Patrulha, sem obrigação de retenção na fonte.
              end
              else if CampoTecno('OptanteSimplesNacional') = '1' then
              begin
                if CampoTecno('ISSRetido') = '2' then
                  SetarCampoValor('NaturezaOperacao', '59') // 5.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional, sem retenção na fonte;
                else
                  SetarCampoValor('NaturezaOperacao', '56'); // 5.6 - Imposto recolhido pelo regime único de arrecadação Simples Nacional, com retenção na fonte;
              end
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '58') // 5.8 - Não Tributável.
              else if CampoTecno('NaturezaTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '57'); // 5.7 - ISSQN Fixo.
            end
            else
            // [6.0] Prestação de serviços para outros municípos da federação.
            begin
              if CampoTecno('MunicipioIncidencia') = _CodigoCidadeConfig then
              begin
                if CampoTecno('ISSRetido') = '1' then
                  SetarCampoValor('NaturezaOperacao', '61') // 6.1 - Imposto devido em  Santo Antônio da Patrulha, com obrigação de retenção na fonte.
                else
                  SetarCampoValor('NaturezaOperacao', '62'); // 6.2 - Imposto devido em Santo Antônio da Patrulha, sem obrigação de retenção na fonte.
              end
              else if CampoTecno('TipoTributacao') = '4' then
                SetarCampoValor('NaturezaOperacao', '68') // 6.8 - Não tributável.
              else if CampoTecno('OptanteSimplesNacional') = '1' then
              begin
                if CampoTecno('ISSRetido') = '2' then
                  SetarCampoValor('NaturezaOperacao', '69') // 6.9 - Imposto recolhido pelo regime único de arrecadação Simples Nacional, sem retenção na fonte;
                else
                  SetarCampoValor('NaturezaOperacao', '66'); // 6.6 - Imposto recolhido pelo regime único de arrecadação Simples Nacional, com retenção na fonte;
              end
              else if CampoTecno('NaturezaTributacao') = '2' then
                SetarCampoValor('NaturezaOperacao', '67') // 6.7 - ISSQN Fixo.
              else if IssRetido then
                SetarCampoValor('NaturezaOperacao', '63') // 6.3 - Imposto devido fora de Santo Antônio da Patrulha, com obrigação de retenção na fonte.
              else
                SetarCampoValor('NaturezaOperacao', '64'); // 6.4 - Imposto devido fora de Santo Antônio da Patrulha, sem obrigação de retenção na fonte.
            end;

            if CampoTecno('PaisTomador') <> '' then
            begin
              if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '01058') and (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') then
                SetarCampoValor('NaturezaOperacao', '78') // 7.8 - Não tributável, prestação no exterior.
            else if (CampoTecno('PaisTomador') <> '1058') and (CampoTecno('PaisTomador') <> '010158') and  (UpperCase(CampoTecno('PaisTomador')) <> 'BRASIL') 
              and (CampoTecno('OptanteSimplesNacional') = '1') then
                SetarCampoValor('NaturezaOperacao', '79') // 7.9 Imposto recolhido pelo regime único de arrecadação.
            end;
          end;
        end;

        _ValorCampo := CampoTecno('RegimeEspecialTributacao');
        if (_ValorCampo <> '') and (_ValorCampo <> '0') then
        begin
          if (AnsiMatchText(_ValorCampo, ['1', '2', '3', '4', '5', '6'])) then
            SetarCampo('RegimeEspecialTributacao')
          else
            RaiseException('Valor inválido para o campo RegimeEspecialTributacao. Valores aceitos: 1, 2, 3, 4, 5, 6.');        
        end;

        _ValorCampo := RemoveMascara(CampoTecno('NumeroRpsSubstituido'));
        VerificaSeEhNumero('NumeroRpsSubstituido', _ValorCampo);
        SetarCampoValorTamanho('NumeroRpsSubstituido', _ValorCampo, 15);

        SetarCampoTamanho('SerieRpsSubstituido', 'SerieRpsSubstituido', 5);

        _ValorCampo := CampoTecno('TipoRpsSubstituido');
        if AnsiMatchText(_ValorCampo, ['', '1', '2', '3']) then
          SetarCampoValorTamanho('TipoRpsSubstituido', _ValorCampo, 1)
        else
          RaiseException('Para essa cidade o campo TipoRpsSubstituido deve ser informado da seguinte forma: 1 - RPS; 2 - Nota Fiscal Conjugada (Mista); 3 – Cupom.');

        SetarCampoCurrency('ValorServicos', True);
        SetarCampoCurrency('ValorDeducoes', False);
        SetarCampoCurrency('ValorPis', False);
        SetarCampoCurrency('ValorCofins', False);
        SetarCampoCurrency('ValorInss', False);
        SetarCampoCurrency('ValorIr', False);
        SetarCampoCurrency('ValorCsll', False);

        case AnsiIndexText(CampoTecno('IssRetido'), ['', '1', 'sim', 'true', '2', 'nao', 'não', 'false']) of
          0: // Vazio: deduzir do TipoTributacao
          begin
            if CampoTecno('TipoTributacao') = '5' then
              SetarCampoValor('IssRetido', '1')
            else
              SetarCampoValor('IssRetido', '2');
          end;
          1, 2, 3: SetarCampoValor('IssRetido', '1');
          4, 5, 6, 7: SetarCampoValor('IssRetido', '2');
        else
          RaiseException('Valor inválido para o campo IssRetido. Valores aceitos: 1, sim, true, 2, nao, não, false.');
        end;

        SetarCampoCurrency('ValorIss', False);

        if CampoTecno('ValorIssRetido') <> '' then
          SetarCampoCurrency('ValorIssRetido', False)
        else
        begin
          // Deduzível (se IssRetido=Sim, utiliza ValorIss)
          if AnsiMatchText(CampoTecno('IssRetido'), [ '1', 'sim', 'true']) then
            SetarCampoValor('ValorIssRetido', CampoCidade('ValorIss'));
        end;

        SetarCampoCurrency('OutrasRetencoes', False);
        SetarCampoCurrency('BaseCalculo', False);
        SetarCampoCurrency('ValorLiquidoNfse', False);
        SetarCampoCurrency('DescontoIncondicionado', False);
        SetarCampoCurrency('DescontoCondicionado', False);

        // Transformável: para Thema, 5% = 0.05
        if DivideCampoPorcentagem(CampoTecno('AliquotaISS')) then
        begin
          _Vlr := StrToFloat(StringReplace(CampoTecno('AliquotaISS'),',','.',[rfReplaceAll])) / 100;  // 5 --> 0,05
          _ValorCampo := FormataFloat(_Vlr, 5, 4);
          SetarCampoValor('Aliquota', _ValorCampo);
        end
        else
          RaiseException('Valor inválido para o campo AliquotaISS.');

        _ValorCampo := FormatarCodigoItemListaServico(CampoTecno('CodigoItemListaServico'), False);  // São Leopoldo: sem ponto no ItemListaServico.
        SetarCampoValorTamanhoObrigatorio('ItemListaServico', _ValorCampo, 5);

        SetarCampoValorTamanho('CodigoCnae', CampoTecno('CodigoCnae'), 7);
        SetarCampoValorTamanho('CodigoTributacaoMunicipio', CampoTecno('CodigoTributacaoMunicipio'), 20);

        SetarCampoValorTamanhoObrigatorio('Discriminacao', CampoTecno('DiscriminacaoServico'), 2000);

        _ValorCampo := ConverterCodigoMunicipio(CampoTecno('CodigoCidadePrestacao'), True);
        SetarCampoValorTamanhoObrigatorio('CodigoMunicipio', _ValorCampo, 7);
        
        _ValorCampo := RemoveMascara(CampoTecno('CpfCnpjPrestador'));
        SetarCpfCnpj('CnpjPrestador', 'CnpjPrestador', 'CpfCnpjPrestador', _ValorCampo);
        SetarCampoValorTamanho('InscricaoMunicipalPrestador', RemoveMascara(CampoTecno('InscricaoMunicipalPrestador')), 15);// TODO: ver nº digitos

        _ValorCampo := RemoveMascara(CampoTecno('CpfCnpjTomador'));
        case Length(_ValorCampo) of
           0: ; //IGNORADO
          11: SetarCampoValor('CpfTomador', _ValorCampo);
          14: SetarCampoValor('CnpjTomador', _ValorCampo);
        else
          RaiseException('Valor inválido para o campo CpfCnpjTomador. CPF: 11 Caracteres; CNPJ: 14 Caracteres.');
        end;
        
        SetarCampoValorTamanho('RazaoSocialTomador', CampoTecno('RazaoSocialTomador'), 115);
        SetarCampoValorTamanho('InscricaoMunicipalTomador', RemoveMascara(CampoTecno('InscricaoMunicipalTomador')), 15);// TODO: ver nº digitos
        SetarCampoValorTamanho('EnderecoTomador', CampoTecno('EnderecoTomador'), 125);
        SetarCampoValorTamanho('NumeroTomador', CampoTecno('NumeroTomador'), 10);
        SetarCampoValorTamanho('ComplementoTomador', CampoTecno('ComplementoTomador'), 60);
        SetarCampoValorTamanho('BairroTomador', CampoTecno('BairroTomador'), 60);

        _ValorCampo := ConverterCodigoMunicipio(CampoTecno('CodigoCidadeTomador'), True);
        SetarCampoValorTamanho('CodigoMunicipioTomador', _ValorCampo, 7);

        if CampoTecno('UfTomador') <> '' then
          SetarCampo('UfTomador')
        else
        begin
          _ValorCampo := BuscarInfoCidade('CodIBGE', CampoCidade('CodigoMunicipioTomador'), 'UF');
          SetarCampoValor('UfTomador', _ValorCampo);
        end;

        SetarCampoValorTamanho('CepTomador', CampoTecno('CepTomador'), 8);
        VerificaDDD('TelefoneTomador', 'TelefoneTomador', 'DDDTomador');
        SetarCampoValorTamanho('EmailTomador', CampoTecno('EmailTomador'), 80);

        SetarCampoValorTamanho('RazaoSocialIntermediario', CampoTecno('RazaoSocialIntermediario'), 115);
        _ValorCampo := RemoveMascara(CampoTecno('CpfCnpjIntermediario'));
        case Length(_ValorCampo) of
          0 : { Campo vazio, ignorar };
          11: SetarCampoValor('CpfIntermediario', _ValorCampo);
          14: SetarCampoValor('CnpjIntermediario', _ValorCampo);
        else
          RaiseException('Valor inválido para o campo CpfCnpjIntermediario');
        end;
        
        SetarCampoValorTamanho('InscricaoMunicipalIntermediario', RemoveMascara(CampoTecno('InscricaoMunicipalIntermediario')), 15);// TODO: ver nº digitos

        SetarCampoValorTamanho('CodigoObra', CampoTecno('CodigoObra'), 15);
        SetarCampoValorTamanho('Art', CampoTecno('Art'), 15);

        _DsCidadeAtual.Post;

        _DsTecnoAtual.Next;
      end;

      SetarCamposMunicipio(false);
      // Atribuir os campos do lote
      Progress('Usando Padrão TecnoNFSe: carregando dados do lote');
      _DsTecnoAtual := TecnoNFSeDS.GetDataSetByName(DSTECNO_LOTE);
      _DsCidadeAtual := Datasets.GetDataSetByName(DSCIDADE_LOTE);

      // IdLote é deduzível do NumeroLote
      if CampoTecno('IdLote') <> '' then
        SetarCampo('IdLote')
      else
      begin
        _ValorCampo := 'LOTE_' + CampoTecno('NumeroLote');
        SetarCampoValor('IdLote', _ValorCampo);
      end;

      SetarCampoValorTamanhoObrigatorio('NumeroLote', CampoTecno('NumeroLote'), 15);
      if CampoTecno('CpfCnpjRemetente') <> '' then
        SetarCampoValor('Cnpj', RemoveMascara(CampoTecno('CpfCnpjRemetente')))
      else
        SetarCampoValor('Cnpj', RemoveMascara(TecnoNFSeDS.GetDataSetByName(DSTECNO_NOTA).FieldByName('CpfCnpjPrestador').AsString));

      _ValorCampo := CampoTecno('InscricaoMunicipalRemetente');
      if _ValorCampo = '' then
        _ValorCampo := TecnoNFSeDS.GetDataSetByName(DSTECNO_NOTA).FieldByName('InscricaoMunicipalPrestador').AsString;
      SetarCampoValorTamanho('InscricaoMunicipal', RemoveMascara(_ValorCampo), 15);

      if CampoTecno('QuantidadeRps') <> '' then
        SetarCampo('QuantidadeRps')
      else
        SetarCampoValor('QuantidadeRps', IntToStr(_QtdeRps));

      Datasets.GetDataSetByName(DSCIDADE_LOTE).Post;
    end
    else
    begin
      Datasets.LoadFromTx2Content(_Tx2.Text);
    end;
  finally
    _Tx2.Free;
    _CamposMunicipio.Free;
    Progress('Arquivo TX2 carregado com Sucesso');
  end;
end.
